function isoToIt(iso){ if(!iso)return''; const[y,m,d]=iso.split('-'); return`${d}/${m}/${y}`; }

function computeStatus(row){

  if(!row.scadenzaISO)return'ok';

  const d=new Date(row.scadenzaISO+'T00:00:00'); if(isNaN(d))return'ok';

  const today=new Date(); today.setHours(0,0,0,0);

  const diff=Math.floor((d-today)/86400000);

  if(diff<0)return'overdue'; if(diff<=Number(row.preavviso??0))return'due'; return'ok';

}

function isPastDate(iso){ if(!iso)return false; const d=new Date(iso+'T00:00:00'); const t=new Date(); t.setHours(0,0,0,0); return d<t; }

function calcolaProssimaScadenza(iso,ricorrenza,ogniX){

  const d=new Date(iso+'T00:00:00'); const r=String(ricorrenza||'NESSUNA').toUpperCase(); const x=Number(ogniX)||1;

  if(r==='NESSUNA')return iso;

  if(r==='GIORNALIERA'){d.setDate(d.getDate()+1);}

  else if(r==='X_GIORNI'){d.setDate(d.getDate()+x);}

  else if(r==='SETTIMANALE'){d.setDate(d.getDate()+7);}

  else if(r==='X_SETTIMANE'){d.setDate(d.getDate()+x*7);}

  else if(r==='MENSILE'){addMesiSafe(d,1);}

  else if(r==='X_MESI'){addMesiSafe(d,x);}

  else if(r==='ANNUALE'){addMesiSafe(d,12);}

  return d.toISOString().split('T')[0];

}
